```cpp
struct list_node{
    int val;
    list_node* next;
    list_node* pre;
    list_node():val(-1),next(nullptr),pre(nullptr){};
    list_node(int x):val(x),next(nullptr),pre(nullptr){};
};
//伪头结点，末尾结点
list_node* head;
list_node* tail;
void init_list_node(){
    head = new list_node();
    tail = new list_node();
    head->next = tail;
    tail->pre = head;
}
//在pre节点后加入一个结点
list_node* insert(list_node* pre, list_node* t){
    auto post = pre->next;
    t->next = post;
    t->pre = pre;
    pre->next = t;
    post->pre = t;
    return t;
}
list_node* insert(list_node* pre, int x){
    list_node* t = new list_node(x);
    insert(pre, t);
    return t;
}
//在post节点前加入一个结点
list_node* insert_pre(list_node* post, list_node* t){
    auto pre = post->pre;
    pre = insert(pre,t);
    return t;
}
list_node* insert_pre(list_node* post, int x){
    auto pre = post->pre;
    insert(pre,x);
    return post->pre;
}
//删除当前节点
void del(list_node* p){
    auto pre = p->pre;
    auto post = p->next;
    pre->next = post;
    post->pre = pre;
}

list_node* addhead(int x){
    insert(head, x);
    return head->next;
}
list_node* addhead(list_node* t){
    insert(head,t);
    return t;
}
```

```cpp
class UnionFind{
private: vector<int> fa;
public:
    UnionFind(int n){
        fa.resize(n+1);
        for(int i = 0; i <= n; i++)
            fa[i] = i;
    }
    int find(int x){
        if( x == fa[x]) return x;
        return fa[x] = find(fa[x]);
    }
    void merge(int x1,int x2){
        int r1 = find(x1),r2 = find(x2);
        fa[r2] = r1;
    }
    bool ask(int a, int b){
        return find(a) == find(b);
    }   
};
```

#### KMP

```cpp
namespace KMP{
    vector<int> ne;
    void getNext(string& p){
        int m = p.size();
        ne.resize(m);
        ne[0] = 0;
        for(int i = 1; i < m; i++){
            int j = ne[i - 1];//j初始化为前面数组的最后一位
            //找到最后一个相同的
            while (j > 0 && p[i] != p[j]) j = ne[j - 1];
            if (p[i] == p[j]) j++; //可以向右拓展
            ne[i] = j;
        }
    }
    //返回所有匹配的位置
    vector<int> match(string& s, string& p){
        int n = s.size(), m = p.size();
        vector<int> res;
        getNext(p);
        int i = 0, j = 0;
        while(i <n){
            if(s[i] ==p[j]) i++, j++;
            else if(j) j = ne[j-1]; //j != 0
            else i++; //j == 0
            if(j == m){
                res.push_back(i-j);
                j = ne[j-1];
            }
        }
        return res;
    }
}
```

#### 字典树

```cpp
class Trie{
    Trie* next[26] = {nullptr};
    int cnt = 0;
public:    
    Trie(){};
    //插入字典树中
    void insert(string& s){
        Trie* p = this;
        for(int i = 0; i < s.size(); i++){
            int pos = s[i] - 'a';
            if(p->next[pos] == nullptr) p->next[pos] = new Trie();
            p = p->next[pos];
        }
        p->cnt++;
    }
    //统计出现次数
    int search(string& s){
        Trie* p = this;
        for(int i = 0; i < s.size(); i++){
            int pos = s[i] - 'a';
            if(p->next[pos] == nullptr) return false;
            p = p->next[pos];
        }
        return p->cnt;
    }
    bool isprefix(string& prefix) {
        Trie* p = this;
        for(auto& c:prefix){
            if(!p->next[c-'a']) return false;
            p = p->next[c-'a'];
        }
        return true;
    }
};
```

#### 字符串hash

```cpp
class StrHash{
    /*核心思想：将字符串看成P进制数，P的经验值是131或13331，取这两个值的冲突概率低
    小技巧：取模的数用2^64，这样直接用unsigned long long存储，溢出的结果就是取模的结果*/
    typedef unsigned long long ull;
private:
    vector<ull> h, p;
    const int P = 131;
public:
    StrHash(string& s){
        int n = s.size();
        h.resize(n+1), p.resize(n+1);
        p[0] = 1;
        for (int i = 1; i <= n; i++ )
        {
            h[i] = h[i - 1] * P + s[i-1];//s的index begin at 0;
            p[i] = p[i - 1] * P;
        }
    }
    ull get(int l, int r){
        return h[r] - h[l - 1] * p[r - l + 1];
    }
    ull get0(int l, int r){
        return get(l+1,r+1);
    }
    bool issame(int l1, int r1, int l2, int r2){
        return get(l1,r1) == get(l2,r2);
    }
    bool issame0(int l1, int r1, int l2, int r2){
        return issame(l1+1,r1+1,l2+1,r2+1);
    }
};
```
